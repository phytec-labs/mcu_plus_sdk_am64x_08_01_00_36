1119 static inline uint32_t UART_getChar(uint32_t baseAddr, uint8_t *pChar)
1120 {
1121  uint32_t lcrRegValue = 0U;
1122  uint32_t retVal = FALSE;
1123 
1124  /* Preserving the current value of LCR. */
1125  lcrRegValue = HW_RD_REG32(baseAddr + UART_LCR);
1126 
1127  /* Switching to Register Operational Mode of operation. */
1128  HW_WR_REG32(baseAddr + UART_LCR, HW_RD_REG32(baseAddr + UART_LCR)
1129  & 0x7FU);
1130 
1131  /* Checking if the RX FIFO(or RHR) has atleast one byte of data. */
1132  if ((uint32_t) UART_LSR_RX_FIFO_E_RX_FIFO_E_VALUE_0 !=
1133  (HW_RD_REG32(baseAddr + UART_LSR) &
1134  UART_LSR_RX_FIFO_E_MASK))
1135  {
1136  uint32_t tempRetVal = HW_RD_REG32(baseAddr + UART_RHR);
1137  *pChar = (uint8_t)tempRetVal;
1138  retVal = TRUE;
1139  }
1140 
1141  /* Restoring the value of LCR. */
1142  HW_WR_REG32(baseAddr + UART_LCR, lcrRegValue);
1143 
1144  return retVal;
1145 }

1147 static inline void UART_intrEnable(uint32_t baseAddr, uint32_t intrFlag)
1148 {
1149  uint32_t enhanFnBitVal = 0U;
1150  uint32_t lcrRegValue = 0U;
1151 
1152  /* Switch to mode B only when the upper 4 bits of IER needs to be changed */
1153  if ((intrFlag & 0xF0U) > 0U)
1154  {
1155  /* Preserving the current value of LCR. */
1156  lcrRegValue = HW_RD_REG32(baseAddr + UART_LCR);
1157  /* Switching to Register Configuration Mode B. */
1158  HW_WR_REG32(baseAddr + UART_LCR, UART_REG_CONFIG_MODE_B & 0xFFU);
1159 
1160  /* Collecting the current value of EFR[4] and later setting it. */
1161  enhanFnBitVal = HW_RD_FIELD32(baseAddr + UART_EFR, UART_EFR_ENHANCED_EN);
1162 
1163  HW_WR_FIELD32(baseAddr + UART_EFR, UART_EFR_ENHANCED_EN,
1164  UART_EFR_ENHANCED_EN_ENHANCED_EN_U_VALUE_1);
1165 
1166  /* Restoring the value of LCR. */
1167  HW_WR_REG32(baseAddr + UART_LCR, lcrRegValue);
1168 
1169  /* Preserving the current value of LCR. */
1170  lcrRegValue = HW_RD_REG32(baseAddr + UART_LCR);
1171 
1172  /* Switching to Register Operational Mode of operation. */
1173  HW_WR_REG32(baseAddr + UART_LCR, HW_RD_REG32(baseAddr + UART_LCR)
1174  & 0x7FU);
1175 
1176  /*
1177  ** It is suggested that the System Interrupts for UART in the
1178  ** Interrupt Controller are enabled after enabling the peripheral
1179  ** interrupts of the UART using this API. If done otherwise, there
1180  ** is a risk of LCR value not getting restored and illicit characters
1181  ** transmitted or received from/to the UART. The situation is explained
1182  ** below.
1183  ** The scene is that the system interrupt for UART is already enabled
1184  ** and the current API is invoked. On enabling the interrupts
1185  ** corresponding to IER[7:4] bits below, if any of those interrupt
1186  ** conditions already existed, there is a possibility that the control
1187  ** goes to Interrupt Service Routine (ISR) without executing the
1188  ** remaining statements in this API. Executing the remaining statements
1189  ** is critical in that the LCR value is restored in them.
1190  ** However, there seems to be no risk in this API for enabling
1191  ** interrupts corresponding to IER[3:0] because it is done at the end
1192  ** and no statements follow that.
1193  */
1194 
1195  /************* ATOMIC STATEMENTS START *************************/
1196 
1197  /* Programming the bits IER[7:4]. */
1198  HW_WR_REG32(baseAddr + UART_IER, intrFlag & 0xF0U);
1199 
1200  /* Restoring the value of LCR. */
1201  HW_WR_REG32(baseAddr + UART_LCR, lcrRegValue);
1202 
1203  /* Preserving the current value of LCR. */
1204  lcrRegValue = HW_RD_REG32(baseAddr + UART_LCR);
1205  /* Switching to Register Configuration Mode B. */
1206  HW_WR_REG32(baseAddr + UART_LCR, UART_REG_CONFIG_MODE_B & 0xFFU);
1207 
1208  /* Restoring the value of EFR[4] to its original value. */
1209  HW_WR_FIELD32(baseAddr + UART_EFR, UART_EFR_ENHANCED_EN, enhanFnBitVal);
1210 
1211  /* Restoring the value of LCR. */
1212  HW_WR_REG32(baseAddr + UART_LCR, lcrRegValue);
1213 
1214  /************** ATOMIC STATEMENTS END *************************/
1215  }
1216 
1217  /* Programming the bits IER[3:0]. */
1218  HW_WR_REG32(baseAddr + UART_IER, HW_RD_REG32(baseAddr + UART_IER) |
1219  (intrFlag & 0x0FU));
1220 }

1267  HW_WR_FIELD32(baseAddr + UART_EFR, UART_EFR_ENHANCED_EN, enhanFnBitVal);

1271  }
