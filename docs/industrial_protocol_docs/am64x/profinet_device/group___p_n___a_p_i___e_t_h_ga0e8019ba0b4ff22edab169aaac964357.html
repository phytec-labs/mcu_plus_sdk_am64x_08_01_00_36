Example
#include <PN_API_ETH.h>
uint32_t result;
PN_API_ETH_SPermanentDcpData_t dcpData = {0};
APPL_SPermanentData_t *pPermanentData = APPL_getPermStorage ();

result = PN_API_ETH_applyPermanentDcpData (&dcpData);
